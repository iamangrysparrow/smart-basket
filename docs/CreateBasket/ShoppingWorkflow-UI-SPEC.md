### Template Selector

```csharp
public class WorkflowEventTemplateSelector : DataTemplateSelector
{
    public DataTemplate SearchProgressTemplate { get; set; }
    public DataTemplate ProductSelectionTemplate { get; set; }
    public DataTemplate ProductPendingTemplate { get; set; }
    public DataTemplate ProductNotFoundTemplate { get; set; }
    
    public override DataTemplate SelectTemplate(object item, DependencyObject container)
    {
        return item switch
        {
            SearchProgressEvent => SearchProgressTemplate,
            ProductSelectionEvent { IsCompleted: true, Selected: not null } => ProductSelectionTemplate,
            ProductSelectionEvent { IsCompleted: true, Selected: null } => ProductNotFoundTemplate,
            ProductSelectionEvent { IsCompleted: false } => ProductPendingTemplate,
            _ => base.SelectTemplate(item, container)
        };
    }
}
```
